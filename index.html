5. Factorial of a number

//Finding Factorial of a number

#include <iostream>
using namespace std;
int main(){ 
    int n;
    cout<<"Enter number: ";
    cin>>n;
    int fact = 1;
    for(int i=1; i<=n; i++){
        fact=fact*i;
    }
    cout<<"Factorial of n is: "<<fact;
}

9. Finding maximum in arrays

//Finding maximum in arrays

#include <iostream>
using namespace std;
int main(){
    int size;
    cout<<"Enter size of array: ";
    cin>>size;
    int arr[100];
    cout<<"Enter elements of array"<<endl;
    
//Taking input

    for(int i=0; i<size; i++){
        cin>>arr[i];
    }
    
//Printing array

    for(int i=0; i<size; i++){
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    
//Finding maximum

    int max = arr[0];
    for(int i=1; i<size; i++){
        if(arr[i] > max){
            max = arr[i];
        }
    }
    cout<<"Maximum element is: "<<max;
return 0;
}

10. Finding sum of elements in arrays

//Finding sum of elements in arrays

#include <iostream>
using namespace std;
int main(){
    int size;
    cout<<"Enter size of array: ";
    cin>>size;
    int arr[100];
    cout<<"Enter elements of array"<<endl;
    
//Taking input

    for(int i=0; i<size; i++){
        cin>>arr[i];
    }
    
//Printing array

    for(int i=0; i<size; i++){
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    
//Finding sum of elements in array

    int sum=0;
    for(int i=0; i<size; i++){
        sum=sum+arr[i];
    }
    cout<<"Sum of elements in array is: "<<sum;

return 0;
}
